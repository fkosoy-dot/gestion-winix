Schema SQL — Crear tablas
Ejecutá este SQL en el editor de Supabase (SQL Editor → New query) para crear las tablas necesarias.
-- Winix Admin Financiero — Schema
-- Ejecutar en Supabase SQL Editor

create table if not exists movimientos (
  id uuid primary key default gen_random_uuid(),
  tipo text not null check (tipo in ('ingreso','egreso')),
  descripcion text not null,
  categoria text,
  monto numeric(12,2) not null check (monto >= 0),
  fecha date not null,
  producto text check (producto in ('winix20','build','ambos') or producto is null),
  cliente text,
  notas text,
  estado text not null default 'realizado'
    check (estado in ('realizado','proyectado')),
  created_at timestamptz default now()
);

create table if not exists periodicos (
  id uuid primary key default gen_random_uuid(),
  tipo text not null check (tipo in ('ingreso','egreso')),
  descripcion text not null,
  categoria text,
  monto numeric(12,2) not null check (monto >= 0),
  frecuencia text not null default 'mensual'
    check (frecuencia in ('mensual','bimestral','trimestral','semestral','anual','semanal','quincenal')),
  dia integer check (dia between 1 and 31),
  desde date not null,
  hasta date,
  producto text check (producto in ('winix20','build','ambos') or producto is null),
  cliente text,
  notas text,
  activo boolean not null default true,
  created_at timestamptz default now()
);

-- Índices
create index if not exists idx_mov_fecha on movimientos(fecha desc);
create index if not exists idx_mov_tipo on movimientos(tipo);
create index if not exists idx_mov_estado on movimientos(estado);
create index if not exists idx_per_activo on periodicos(activo);
create index if not exists idx_per_tipo on periodicos(tipo);

-- Deshabilitar RLS para uso personal
alter table movimientos disable row level security;
alter table periodicos disable row level security;

-- Verificar
select 'movimientos' as tabla, count(*) from movimientos
union all
select 'periodicos', count(*) from periodicos;
